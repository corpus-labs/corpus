A wallet connect button that integrates with the `@solana/wallet-adapter-react` package.

### Component usage

```js
import { WalletProvider } from '@corpus/features.wallet.provider';
import { WalletConnect } from '@corpus/features.wallet.connect';

<WalletProvider>
  <WalletConnect />
</WalletProvider>
```
